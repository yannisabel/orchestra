import { Canvas, Meta, Story } from '@storybook/addon-docs'

import { Box } from '../layout'
import { OrchestraProvider } from '../providers'
import { Text } from '../text'

import {allColors} from '.'

<Meta title="Instruments/Colors" />

# Colors

<Canvas>
  <Story name="Default">
    <OrchestraProvider>
      {Object.entries(allColors).map(([colorName, value]) => {
        return (
          <Box backgroundColor={colorName} key={`${colorName}`}>
            <Text color={colorName.startsWith('black-') || colorName.startsWith('blue-') ? 'white-0' : 'grey-100'}>
              colorName: {colorName}
            </Text>
            <Text color={colorName === 'black-0' ? 'white-0' : 'grey-100'}>
              value: {value}
            </Text>
          </Box>
        )
      })}
    </OrchestraProvider>
  </Story>
</Canvas>
