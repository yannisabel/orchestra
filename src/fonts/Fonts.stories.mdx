import { Canvas, Meta, Story } from '@storybook/addon-docs'

import { Box } from '../layout'
import { OrchestraProvider } from '../providers'
import { Text } from '../text'

import {fontFamilies, fontSizes, fontWeights} from '.'

<Meta title="Instruments/Fonts" />

# Fonts

## Font Family
<Canvas>
  <Story name="Font Family">
    <OrchestraProvider>
      {Object.entries(fontFamilies).map(([fontName, value]) => {
        return (
          <Box key={`${fontName}`}>
            <Text fontFamily={fontName}>
              fontName: {fontName}
            </Text>
            <Text fontFamily={fontName}>
              value: {value}
            </Text>
          </Box>
        )
      })}
    </OrchestraProvider>
  </Story>
</Canvas>
